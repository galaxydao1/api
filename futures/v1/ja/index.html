import hashlib
import hmac
import json
import time
import requests

API_KEY = '900625568558820892a8c833c33ebc8fd2701efe'
SECRET_KEY = 'c708ac3e70d115ec29efbee197330627d7edf842'
# BASE_URL = 'https://api.bost.com'

def do_sign(body):
    timestamp = int(time.time()) * 1000
    # to_sign = str(timestamp)+json.dumps(body,separators=(',',':'))
    to_sign = str(timestamp) + json.dumps(body)
    sign = hmac.new(SECRET_KEY.encode("utf-8"), to_sign.encode("utf-8"), hashlib.md5).hexdigest()
    print(to_sign)

    headers = {
        'api-key': API_KEY,
        'api-sign': sign,
        'api-timestamp': str(timestamp)
    }
    return headers

        'amount': amount,
    }

    resp = requests.post(BASE_URL + path, json=body, headers=headers)
    print(resp.text)


if __name__ == '__main__':

    • 2.インターフェイスリクエストで必要なパラメータparamを理解し、strParamを取得するようにパラメータをフォーマットします。

    • 3.署名が必要な文字列 strToSign を生成します。

    • 4.apikey のパスワード シークレットを使用して、署名が必要な文字列 strToSign に対して HmacMD5 暗号化署名を実行します。

    • 5.リクエストの際はhttp headers の以下の項目を設定してください

    フィールド名 タイプ 説明 値
    content-type string 'application/json'
    api-key string your apikey
    api-timestamp long 現在のタイムスタンプ (ミリ秒) timestamp
    api-sign string 署名結果 sign

    • 6.リクエスト時はフォーマット済みのリクエストパラメータ strParam に http body を設定してください

    • 7.戻り値にはフィールド state が含まれており、0 は成功を示し、その他のエラー コードは失敗を示します。 成功した結果には対応するフィールド result があり、失敗した結果には対応するフィールド msg があります。

    資金が契約口座に送金された後でのみ、注文を正常に行うことができます。 注文の数量パラメーター amount は、契約の額面の整数倍である必要があります。本来の目的は、1 つの契約を基本単位として使用することです。 例: 4XRP_USDT 契約の額面が 10XRP の場合、amount パラメータは 10 の整数倍でなければなりません。渡された値が 10 の整数倍でない場合は、「amount パラメータ エラー」というメッセージが表示されます。 市場価格注文を行う場合、price パラメータとして 1 を渡すだけで、システムは最良価格に基づいて注文を出します。 最適価格はレバレッジ倍率に関連しており、レバレッジが高い場合、最適価格が現在のインデックス価格から大きく乖離することが制限されるため、注文が完全に完了しない、または完了しない可能性があります。 指値注文を行う場合、price パラメータも最適価格によって制限され、レバレッジが高い場合、注文価格は現在のインデックス価格から大きく乖離することができず、注文が完了するとすぐにポジションが強制決済されます。禁じられている。

    pair true string 契約記号 4BTC_USDT,4ETH_USDT, ...
    amount true string 数量 (契約額面の倍数である必要があります) を文字列に変換します
    order_side true integer 注文タイプ: 1 オープンロング、2 オープンショート、3 クローズロング、4 クローズショート 1,2,3,4
    order_type true integer 注文方法、1は成行注文、2は指値注文を表します 1,2
    price true string 注文価格
    order_from true integer 注文元 6
    client_oid false string カスタマイズされた id
